21  void addOneQubitGate(const std::string name, xacc::Instruction &inst) {
22  auto expr = inst.getBitExpression(0);
23  ss << "quantum::" + name + "(" << inst.getBufferNames()[0] << "["
24  << (expr.empty() ? std::to_string(inst.bits()[0]) : expr) << "]";
25  if (inst.isParameterized() && inst.name() != "Measure") {
26  ss << ", " << inst.getParameter(0).toString();
27  for (int i = 1; i < inst.nParameters(); i++) {
28  ss << ", " << inst.getParameter(i).toString() << "\n";
29  }
30  }
31  ss << ");\n";
32  }

84  void visit(Circuit &circ) override {
85  if (circ.name() == kernelName) {
86  return;
87  }
88  if (circ.name() == "exp_i_theta") {
89  ss << "quantum::exp(" << circ.getBufferNames()[0] << ", "
90  << circ.getArguments()[0]->name << ", " << circ.getArguments()[1]->name
91  << ");\n";
92  } else {
93  // Call a previously-defined QCOR kernel:
94  // In this context, we disable __execute flag around this hence
95  // this sub-kernel will not be submitted.
96  // i.e. only the outer-most kernel will be submitted.
97  // Open a new scope since we use a local var '__cached_execute_flag'
98  ss << "{\n";
99  // Cache the state of the __execute flag
100  ss << "const auto __cached_execute_flag = __execute;\n";
101  // Reset the flag:
102  ss << "__execute = false;\n";
103  for (const auto &arg : circ.getArguments()) {
104  if (arg->name == "__xacc__literal_") {
105  // double nameMEMORYLOC = ...
106  ss << arg->type << " " << arg->name << arg << " = ";
107  // can be int or double
108  if (arg->runtimeValue.keyExists<int>(
109  xacc::INTERNAL_ARGUMENT_VALUE_KEY)) {
110  ss << arg->runtimeValue.get<int>(xacc::INTERNAL_ARGUMENT_VALUE_KEY)
111  << ";\n";
112  } else {
113  ss << arg->runtimeValue.get<double>(
114  xacc::INTERNAL_ARGUMENT_VALUE_KEY)
115  << ";\n";
116  }
117  }
118  }
119  // Add the circuit invocation.
120  ss << circ.name() << "(" << circ.getBufferNames()[0];
121  for (const auto &arg : circ.getArguments()) {
122  if (arg->name.find("__xacc__literal_") != std::string::npos) {
123  ss << ", " << arg->name << arg;
124  } else {
125  ss << ", " << arg->name;
126  }
127  }
128  ss << ")"
129  << ";\n";
130  // Reinstate the __execute flag
131  ss << "__execute = __cached_execute_flag;\n";
132  ss << "}\n";
133  }
134  }
135  void visit(IfStmt &ifStmt) override {}
