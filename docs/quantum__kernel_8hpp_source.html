8 // The QuantumKernel represents the super-class of all qcor
9 // quantum kernel functors. Subclasses of this are auto-generated
10 // via the Clang Syntax Handler capability. Derived classes
11 // provide a destructor implementation that builds up and
12 // submits quantum instructions to the specified backend. This enables
13 // functor-like capability whereby programmers instantiate temporary
14 // instances of this via the constructor call, and the destructor is
15 // immediately called. More advanced usage is of course possible for
16 // qcor developers.
17 //
18 // This class works by taking the Derived type (CRTP) and the kernel function
19 // arguments as template parameters. The Derived type is therefore available for
20 // instantiation within provided static methods on QuantumKernel. The Args...
21 // are stored in a member tuple, and are available for use when evaluating the
22 // kernel. Importantly, QuantumKernel provides static adjoint and ctrl methods
23 // for auto-generating those circuits.
24 //
25 // The Syntax Handler will take kernels like this
26 // __qpu__ void foo(qreg q) { H(q[0]); }
27 // and create a derived type of QuantumKernel like this
28 // class foo : public qcor::QuantumKernel<class foo, qreg> {...};
29 // with an appropriate implementation of constructors and destructors.
30 // Users can then call for adjoint/ctrl methods like this
31 // foo::adjoint(q); foo::ctrl(1, q);
32 template <typename Derived, typename... Args> class QuantumKernel {
33 protected:

82  // Create the Adjoint of this quantum kernel
83  static void adjoint(std::shared_ptr<CompositeInstruction> parent_kernel,
84  Args... args) {
85 
86  // instantiate and don't let it call the destructor
87  Derived derived(args...);
88  derived.disable_destructor = true;
89 
90  // run the operator()(args...) call to get the parent_kernel
91  derived(args...);
92 
93  // get the instructions
94  auto instructions = derived.parent_kernel->getInstructions();
95  std::shared_ptr<CompositeInstruction> program = derived.parent_kernel;
96 
97  // Assert that we don't have measurement
98  if (!std::all_of(
99  instructions.cbegin(), instructions.cend(),
100  [](const auto &inst) { return inst->name() != "Measure"; })) {
101  error(
102  "Unable to create Adjoint for kernels that have Measure operations.");
103  }
104 
105  auto provider = qcor::__internal__::get_provider();
106  std::reverse(instructions.begin(), instructions.end());
107  for (int i = 0; i < instructions.size(); i++) {
108  auto inst = derived.parent_kernel->getInstruction(i);
109  // Parametric gates:
110  if (inst->name() == "Rx" || inst->name() == "Ry" ||
111  inst->name() == "Rz" || inst->name() == "CPHASE" ||
112  inst->name() == "U1" || inst->name() == "CRZ") {
113  inst->setParameter(0, -inst->getParameter(0).template as<double>());
114  }
115  // Handles T and S gates, etc... => T -> Tdg
116  else if (inst->name() == "T") {
117  auto tdg = provider->createInstruction("Tdg", inst->bits());
118  program->replaceInstruction(i, tdg);
119  } else if (inst->name() == "S") {
120  auto sdg = provider->createInstruction("Sdg", inst->bits());
121  program->replaceInstruction(i, sdg);
122  }
123  }
124 
125  // add the instructions to the current parent kernel
126  parent_kernel->addInstructions(instructions);
127 
128  // no measures, so no execute
129  }

131  // Create the controlled version of this quantum kernel
132  static void ctrl(std::shared_ptr<CompositeInstruction> parent_kernel,
133  int ctrlIdx, Args... args) {
134  // instantiate and don't let it call the destructor
135  Derived derived(args...);
136  derived.disable_destructor = true;
137 
138  // run the operator()(args...) call to get the the functor
139  // as a CompositeInstruction (derived.parent_kernel)
140  derived(args...);
141 
142  // Use the controlled gate module of XACC to transform
143  auto tempKernel = qcor::__internal__::create_composite("temp_control");
144  tempKernel->addInstruction(derived.parent_kernel);
145 
146  auto ctrlKernel = qcor::__internal__::create_ctrl_u();
147  ctrlKernel->expand({
148  std::make_pair("U", tempKernel),
149  std::make_pair("control-idx", ctrlIdx),
150  });
151 
152  // std::cout << "HELLO\n" << ctrlKernel->toString() << "\n";
153  for (int instId = 0; instId < ctrlKernel->nInstructions(); ++instId) {
154  parent_kernel->addInstruction(
155  ctrlKernel->getInstruction(instId)->clone());
156  }
157  }
