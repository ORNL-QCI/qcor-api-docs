171 // Create and return a random vector<ScalarType> of the given size
172 // where all elements are within the given range
173 template <typename ScalarType>
174 auto random_vector(const ScalarType l_range, const ScalarType r_range,
175  const std::size_t size) {
176  // Generate a random initial parameter set
177  std::random_device rnd_device;
178  std::mt19937 mersenne_engine{rnd_device()}; // Generates random integers
179  std::uniform_real_distribution<ScalarType> dist{l_range, r_range};
180  auto gen = [&dist, &mersenne_engine]() { return dist(mersenne_engine); };
181  std::vector<ScalarType> vec(size);
182  std::generate(vec.begin(), vec.end(), gen);
183  return vec;
184 }

194 // C++17 python-like enumerate utility function
195 template <typename T, typename TIter = decltype(std::begin(std::declval<T>())),
196  typename = decltype(std::end(std::declval<T>()))>
197 constexpr auto enumerate(T &&iterable) {
198  struct iterator {
199  size_t i;
200  TIter iter;
201  bool operator!=(const iterator &other) const { return iter != other.iter; }
202  void operator++() {
203  ++i;
204  ++iter;
205  }
206  auto operator*() const { return std::tie(i, *iter); }
207  };
208  struct iterable_wrapper {
209  T iterable;
210  auto begin() { return iterator{0, std::begin(iterable)}; }
211  auto end() { return iterator{0, std::end(iterable)}; }
212  };
213  return iterable_wrapper{std::forward<T>(iterable)};
214 }
