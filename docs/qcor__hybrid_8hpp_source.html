138  // Execute the VQE task asynchronously, use provided Optimizer
139  template <typename... Args>
140  Handle execute_async(std::shared_ptr<Optimizer> optimizer,
141  Args... initial_params) {
142  // Get the VQE ObjectiveFunction and set the qreg
143  auto objective =
144  qcor::createObjectiveFunction(OBJECTIVE_NAME, ansatz, observable);
145  objective->set_qreg(q);
146 
147  // Convert input args to a tuple
148  auto init_args_tuple = std::make_tuple(initial_params...);
149 
150  // Set the Optimizer initial parameters
151  std::vector<double> init_params;
152  __internal__::ConvertDoubleLikeToVectorDouble build_up_init_params(
153  init_params);
154  __internal__::tuple_for_each(init_args_tuple, build_up_init_params);
155  optimizer->appendOption(OPTIMIZER_INIT_PARAMS, init_params);
156 
157  // Create the Arg Translator
158  TranslationFunctor<qreg, Args...> arg_translator;
159  if (translation_functor.has_value()) {
160  arg_translator =
161  std::any_cast<TranslationFunctor<qreg, Args...>>(translation_functor);
162  } else {
163  __internal__::TranslationFunctorAutoGenerator auto_gen;
164  arg_translator = auto_gen(q, std::tuple<Args...>());
165  }
166 
167  // Count all rotation angles (n parameters)
168  __internal__::CountRotationAngles count_params(n_params);
169  __internal__::tuple_for_each(init_args_tuple, count_params);
170 
171  if (optimizer->isGradientBased()) {
172 
173  if (!grad_eval) {
174  grad_eval = [&, arg_translator, objective](const std::vector<double> x,
175  std::vector<double> &dx) {
176  for (int i = 0; i < dx.size(); i++) {
177  auto xplus = x[i] + pi / 2.;
178  auto xminus = x[i] - pi / 2.;
179  std::vector<double> tmpx_plus = x, tmpx_minus = x;
180  tmpx_plus[i] = xplus;
181  tmpx_minus[i] = xminus;
182 
183  auto translated_tuple_xp = arg_translator(tmpx_plus);
184  auto translated_tuple_xm = arg_translator(tmpx_minus);
185 
186  auto results1 =
187  qcor::__internal__::call(objective, translated_tuple_xp);
188  auto results2 =
189  qcor::__internal__::call(objective, translated_tuple_xm);
190 
191  dx[i] = 0.5 * (results1 - results2);
192  }
193  };
194  }
195 
196  return qcor::taskInitiate(objective, optimizer, grad_eval, arg_translator,
197  n_params);
198 
199  } else {
200  // Run TaskInitiate, kick of the VQE job asynchronously
201  return qcor::taskInitiate(objective, optimizer, arg_translator, n_params);
202  }
203  }
