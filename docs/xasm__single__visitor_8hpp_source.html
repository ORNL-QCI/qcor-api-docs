25  antlrcpp::Any visitQinst(xasm_singleParser::QinstContext *context) override {
26  if (!xacc::isInitialized()) {
27  xacc::Initialize();
28  }
29  // if not in instruction registry, then forward to classical instructions
30  auto inst_name = context->inst_name->getText();
31  auto provider = xacc::getIRProvider("quantum");
32 
33  if (common_name_map.count(inst_name)) {
34  inst_name = common_name_map[inst_name];
35  }
36 
37  if (xacc::container::contains(provider->getInstructions(), inst_name)) {
38 
39  // We don't really care about Instruction::bits(), qrt_mapper
40  // will look for bit expressions and use those, so just set
41  // everything as a string...
42 
43  // Create an instance of the Instruction with the given name
44  auto inst = provider->createInstruction(inst_name, 0);
45 
46  // If it is not composite, look for its bit expressions
47  // and parameter expressions
48  if (!inst->isComposite()) {
49  // Get the number of required bits and parameters
50  auto required_bits = inst->nRequiredBits();
51  auto required_params = inst->getParameters().size();
52 
53  if (required_bits + required_params !=
54  context->explist()->exp().size() &&
55  inst_name != "Measure") {
56  std::stringstream xx;
57  xx << "Invalid quantum instruction expression. " << inst_name
58  << " requires " << required_bits << " qubit args and "
59  << required_params << " parameter args.";
60  xacc::error(xx.str());
61  }
62 
63  // Get the qubit expresssions
64  std::vector<std::string> buffer_names;
65  for (int i = 0; i < required_bits; i++) {
66  auto bit_expr = context->explist()->exp(i);
67  auto bit_expr_str = bit_expr->getText();
68 
69  auto found_bracket = bit_expr_str.find_first_of("[");
70  if (found_bracket != std::string::npos) {
71  auto buffer_name = bit_expr_str.substr(0, found_bracket);
72  auto bit_idx_expr = bit_expr_str.substr(
73  found_bracket + 1, bit_expr_str.length() - found_bracket - 2);
74 
75  buffer_names.push_back(buffer_name);
76  inst->setBitExpression(i, bit_idx_expr);
77  } else {
78  xacc::error("Must provide qreg[IDX] and not just qreg.");
79  }
80  }
81  inst->setBufferNames(buffer_names);
82 
83  // Get the parameter expressions
84  int counter = 0;
85  for (int i = required_bits; i < context->explist()->exp().size(); i++) {
86  inst->setParameter(counter, context->explist()->exp(i)->getText());
87  counter++;
88  }
89  } else {
90 
91  // this is something like exp_i_theta(q,...);
92  auto comp_inst = xacc::ir::asComposite(inst);
93  inst->setBufferNames({context->explist()->exp(0)->getText()});
94  for (int i = 1; i < context->explist()->exp().size(); i++) {
95  comp_inst->addArgument(context->explist()->exp(i)->getText(), "");
96  }
97  }
98 
99  // std::cout << "INST: " << inst->toString() << "\n";
100 
101  result.second = inst;
102  } else {
103  // if we don't know the instruction, then we should just
104  // add it as a classical call, let the compiler throw an
105  // error if it doesn't know it
106  std::stringstream ss;
107  for (auto c : context->children) {
108  ss << c->getText() << " ";
109  }
110  ss << "\n";
111 
112  result.first = ss.str();
113  }
114 
115  return 0;
116  }
